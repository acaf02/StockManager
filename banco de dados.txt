-- Tabela de Login
CREATE TABLE login (
    cod_funcionario INT PRIMARY KEY,         -- Código do funcionário
    login_funcionario VARCHAR(50) NOT NULL,  -- Login do funcionário
    senha VARCHAR(50) NOT NULL,              -- Senha
    nome VARCHAR(100) NOT NULL               -- Nome do funcionário
);

-- Tabela de Insumo
CREATE TABLE insumo (
    cod_insumo INT PRIMARY KEY,              -- Código do insumo
    nome VARCHAR(100) NOT NULL,              -- Nome do insumo
    peso DECIMAL(10, 2) NOT NULL,            -- Peso do insumo
    unidade VARCHAR(10) NOT NULL,            -- Unidade de medida (ex: kg, g, l)
    quantidade DECIMAL(10, 2) NOT NULL,      -- Quantidade disponível
    estoque_min DECIMAL(10, 2) NOT NULL,     -- Estoque mínimo
    estoque_max DECIMAL(10, 2) NOT NULL,     -- Estoque máximo
    consumo_insumo DECIMAL(10, 2) NOT NULL   -- Quantidade consumida do insumo
);
